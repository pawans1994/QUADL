 21 function CTATStringUtil ()
 22 {	
 23 	/**
 24 	* Apparently AS3 expects a direct conversion between a string and a boolean to only
 25 	* happen when the string is '0' or '1' for every other case we have to test manually.
 26 	* So this method pretty much catches all varieties.
 27 	*/	
 28 	this.String2Boolean=function String2Boolean (aString)
 29 	{
 30 		switch(aString) 
 31 		{     
 32 			case "1":     
 33 			case "true":     
 34 			case "yes":     
 35 			case "TRUE":     
 36 			case "YES":
 37 			case "Yes":					
 38 						return true;     
 39 			case "0":     
 40 			case "false":     
 41 			case "no":
 42 			case "FALSE":     
 43 			case "NO":
 44 			case "No":
 45 						return false;						
 46 		}
 47 	
 48 		return (true);		
 49 	}
 50 	/**
 51 	*	
 52 	*/
 53 	this.replaceString=function replaceString (thisStr,searchStr,replaceStr) 
 54 	{	
 55 		return (thisStr.replace(searchStr,replaceStr));
 56 	};	
 57 }
