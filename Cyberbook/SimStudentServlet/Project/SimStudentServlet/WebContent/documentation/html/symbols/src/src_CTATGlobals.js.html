152  */
153 function componentToHex(c) 
154 {
155     var hex = c.toString(16);
156     return hex.length == 1 ? "0" + hex : hex;
157 }

171 /**
172  * 
173  * @param hex
174  * @returns
175  */
176 function hexToRgb(hex) 
177 {
178     var result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
179     
180     return result ? 
181 	{    	
182         r: parseInt(result[1], 16),
183         g: parseInt(result[2], 16),
184         b: parseInt(result[3], 16)
185     } : null;
186 }

193 function loadjscssfile(filename, filetype)
194 {
195 	if (filetype=="js")
196 	{
197 		//if filename is a external JavaScript file
198 		var fileref=document.createElement('script');
199 		fileref.setAttribute("type","text/javascript");
200 		fileref.setAttribute("src", filename);
201 	}
202 	else if (filetype=="css")
203 	{ 
204 		//if filename is an external CSS file
205 		var fileref=document.createElement("link");
206 		fileref.setAttribute("rel", "stylesheet");
207 		fileref.setAttribute("type", "text/css");
208 		fileref.setAttribute("href", filename);
209 	}
210 	
211 	if (typeof fileref!="undefined")
212 	{
213 		document.getElementsByTagName("head")[0].appendChild(fileref);
214 	}	
215 }

217 /**
218 *
219 */
220 function parseQueryString () 
221 {
222 	var str = location.search;
223 		
224 	var query = str.charAt(0) == '?' ? str.substring(1) : str;
225 	var args = new Object();
226 
227 	if (query) 
228 	{
229 		var fields = query.split('&');
230 
231 		for (var f = 0; f < fields.length; f++) 
232 		{
233 			var field = fields[f].split('=');
234 			args[unescape(field[0].replace(/\+/g, ' '))] = unescape(field[1].replace(/\+/g, ' '));
235 		}
236 	}
237 	
238 	return args;
239 }

349 function raiseInitialScrim(aMessage) 
350 {
351 	var mainCanvas=document.getElementById("main-canvas");
352 	mainCanvas.setAttribute('style', "background-color: rgba(0,0,0,0.2);");
353 	
354 	ctx=mainCanvas.getContext("2d");
355 	
356 	var scrimGraphics=new CTATGraphicsTools(ctx);
357 
358 	scrimGraphics.setLineColor("dark grey");
359 	scrimGraphics.setFillColor("white");
360 	
361 	var x=mainCanvas.width * 1/8;
362 	var y=mainCanvas.height * 3/8;
363 	
364 	var w=mainCanvas.width * 6/8;
365 	var h=mainCanvas.height * 3/8;
366 	
367 	scrimGraphics.drawRoundedRectFilled(x, y, w, h, 5);
368 	
369 	scrimGraphics.setLineColor("black");
370 	scrimGraphics.drawText(x, y, aMessage);
371 }

399 /**
400  * 
401  * @param anOrientation
402  */
403 function setOrientation(anOrientation)
404 {
405 	var getDebugger = new CTATBase("tutor", "");
406 	getDebugger.debug ("setOrientation ("+anOrientation+")");		
407 	
408 	switch (anOrientation) 
409 	{  
410 		case -90:
411 		case 90:
412 					orientation='landscape';
413 			
414 					var orImg=document.getElementById("pageor");
415 					if (orImg!=null)
416 					{
417 						orImg.src="skindata/landscape.png";
418 					}
419 					else
420 						alert ("Can't find orientation image object");
421 			
422 					if (mobileAPI!=null)
423 					{
424 						mobileAPI.processOrientationChange (orientation);
425 					}
426 					
427 					break;
428 		case 180:
429 		default:    	
430 					orientation='portrait';
431 
432 					var orImg=document.getElementById("pageor");
433 					if (orImg!=null)
434 					{
435 						orImg.src="skindata/portrait.png";
436 					}
437 					else
438 						alert ("Can't find orientation image object");					
439 			
440 					if (mobileAPI!=null)
441 					{
442 						mobileAPI.processOrientationChange (orientation);
443 					}
444 					
445 					break;
446 	}	
447 }

492 /**
493 * Introspects an object.
494 * http://www.syger.it/Tutorials/JavaScriptIntrospector.html
495 * 
496 * @param name the object name.
497 * @param obj the object to introspect.
498 * @param indent the indentation (optional, defaults to "").
499 * @param levels the introspection nesting level (defaults to 1).
500 * @returns a plain text analysis of the object.
501 */
502 function introspect (name, obj, indent, levels) 
503 {
504  indent = indent || "";
505  
506  if (this.typeOf(levels) !== "number") levels = 1;
507  var objType = this.typeOf(obj);
508  var result = [indent, name, " ", objType, " :"].join('');
509  if (objType === "object") {
510    if (level > 0) {
511      indent = [indent, "  "].join('');
512      for (prop in obj) {
513        var prop = this.introspect(prop, obj[prop], indent, level - 1);
514        result = [result, "\n", prop].join('');
515      }
516      return result;
517    }
518    else {
519      return [result, " ..."].join('');
520    }
521  }
522  else if (objType === "null") {
523    return [result, " null"].join('');
524  }
525  return [result, " ", obj].join('');
