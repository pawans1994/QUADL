263  	this.htmlEncode=function htmlEncode (value)
264  	{
265  		//create a in-memory div, set it's inner text(which jQuery automatically encodes)
266  		//then grab the encoded contents back out.  The div never exists on the page.
267  		return $('<div/>').text(value).html();
268  	};

270  	this.htmlDecode=function htmlDecode (value)
271  	{
272  		return $('<div/>').html(value).text();
