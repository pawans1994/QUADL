 21 function s4() 
 22 {
 23   return Math.floor((1 + Math.random()) * 0x10000)
 24              .toString(16)
 25              .substring(1);
 26 };

 31 function guid() 
 32 {
 33   return s4() + s4() + '-' + s4() + '-' + s4() + '-' +
 34          s4() + '-' + s4() + s4() + s4();
 35 }

 40 Object.size = function(obj) 
 41 {
 42     var size = 0, key;
 43     
 44     for (key in obj) 
 45     {
 46         if (obj.hasOwnProperty(key)) size++;
 47     }
 48     return size;
 49 };

 53 */
 54 function htmlEscape(str) 
 55 {
 56     return String(str)
 57             .replace(/&/g, '&')
 58             .replace(/"/g, '"')
 59             .replace(/'/g, ''')
 60             .replace(/</g, '<')
 61             .replace(/>/g, '>');
 62 }
 63 
 64 /**
 65 *
 66 */
 67 function htmlUnescape(value)
 68 {
 69     return String(value)
 70         .replace(/"/g, '"')
 71         .replace(/'/g, "'")
 72         .replace(/</g, '<')
 73         .replace(/>/g, '>')
 74         .replace(/&/g, '&');
 75 }
