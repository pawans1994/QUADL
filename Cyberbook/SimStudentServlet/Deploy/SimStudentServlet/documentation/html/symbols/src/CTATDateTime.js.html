  9  * Accepts a date, a mask, or a date and a mask.
 10  * Returns a formatted version of the given date.
 11  * The date defaults to the current date/time.
 12  * The mask defaults to dateFormat.masks.default.
 13  */
 14 
 15 var dateFormat = function () 
 16 {
 17 	var	token = /d{1,4}|m{1,4}|yy(?:yy)?|([HhMsTt])\1?|[LloSZ]|"[^"]*"|'[^']*'/g,
 18 		timezone = /\b(?:[PMCEA][SDP]T|(?:Pacific|Mountain|Central|Eastern|Atlantic) (?:Standard|Daylight|Prevailing) Time|(?:GMT|UTC)(?:[-+]\d{4})?)\b/g,
 19 		timezoneClip = /[^-+\dA-Z]/g,
 20 		
 21 	pad = function (val, len) 
 22 	{
 23 		val = String(val);
 24 		len = len || 2;
 25 		while (val.length < len) val = "0" + val;
 26 		return val;
 27 	};
 28 
 29 	/**
 30 	 * Regexes and supporting functions are cached through closure
 31 	 */
 32 	return function (date, mask, utc) 
 33 	{
 34 		var dF = dateFormat;
 35 
 36 		// You can't provide utc if you skip other args (use the "UTC:" mask prefix)
 37 		if (arguments.length == 1 && Object.prototype.toString.call(date) == "[object String]" && !/\d/.test(date)) {
 38 			mask = date;
 39 			date = undefined;
 40 		}
 41 
 42 		// Passing date through Date applies Date.parse, if necessary
 43 		date = date ? new Date(date) : new Date;
 44 		if (isNaN(date)) throw SyntaxError("invalid date");
 45 
 46 		mask = String(dF.masks[mask] || mask || dF.masks["default"]);
 47 
 48 		// Allow setting the utc argument via the mask
 49 		if (mask.slice(0, 4) == "UTC:") {
 50 			mask = mask.slice(4);
 51 			utc = true;
 52 		}
 53 
 54 		var	_ = utc ? "getUTC" : "get",
 55 			d = date[_ + "Date"](),
 56 			D = date[_ + "Day"](),
 57 			m = date[_ + "Month"](),
 58 			y = date[_ + "FullYear"](),
 59 			H = date[_ + "Hours"](),
 60 			M = date[_ + "Minutes"](),
 61 			s = date[_ + "Seconds"](),
 62 			L = date[_ + "Milliseconds"](),
 63 			o = utc ? 0 : date.getTimezoneOffset(),
 64 			flags = {
 65 				d:    d,
 66 				dd:   pad(d),
 67 				ddd:  dF.i18n.dayNames[D],
 68 				dddd: dF.i18n.dayNames[D + 7],
 69 				m:    m + 1,
 70 				mm:   pad(m + 1),
 71 				mmm:  dF.i18n.monthNames[m],
 72 				mmmm: dF.i18n.monthNames[m + 12],
 73 				yy:   String(y).slice(2),
 74 				yyyy: y,
 75 				h:    H % 12 || 12,
 76 				hh:   pad(H % 12 || 12),
 77 				H:    H,
 78 				HH:   pad(H),
 79 				M:    M,
 80 				MM:   pad(M),
 81 				s:    s,
 82 				ss:   pad(s),
 83 				l:    pad(L, 3),
 84 				L:    pad(L > 99 ? Math.round(L / 10) : L),
 85 				t:    H < 12 ? "a"  : "p",
 86 				tt:   H < 12 ? "am" : "pm",
 87 				T:    H < 12 ? "A"  : "P",
 88 				TT:   H < 12 ? "AM" : "PM",
 89 				Z:    utc ? "UTC" : (String(date).match(timezone) || [""]).pop().replace(timezoneClip, ""),
 90 				o:    (o > 0 ? "-" : "+") + pad(Math.floor(Math.abs(o) / 60) * 100 + Math.abs(o) % 60, 4),
 91 				S:    ["th", "st", "nd", "rd"][d % 10 > 3 ? 0 : (d % 100 - d % 10 != 10) * d % 10]
 92 			};
 93 
 94 		return mask.replace(token, function ($0) {
 95 			return $0 in flags ? flags[$0] : $0.slice(1, $0.length - 1);
 96 		});
 97 	};
 98 }();
 99 
100 // Some common format strings
101 dateFormat.masks = 
102 {
103 	"default":      "ddd mmm dd yyyy HH:MM:ss",
104 	shortDate:      "m/d/yy",
105 	mediumDate:     "mmm d, yyyy",
106 	longDate:       "mmmm d, yyyy",
107 	fullDate:       "dddd, mmmm d, yyyy",
108 	shortTime:      "h:MM TT",
109 	mediumTime:     "h:MM:ss TT",
110 	longTime:       "h:MM:ss TT Z",
111 	isoDate:        "yyyy-mm-dd",
112 	isoTime:        "HH:MM:ss",
113 	isoDateTime:    "yyyy-mm-dd'T'HH:MM:ss",
114 	isoUtcDateTime: "UTC:yyyy-mm-dd'T'HH:MM:ss'Z'"
115 };
116 
117 // Internationalization strings
118 dateFormat.i18n = 
119 {
120 	dayNames: [
121 		"Sun", "Mon", "Tue", "Wed", "Thu", "Fri", "Sat",
122 		"Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"
123 	],
124 	monthNames: [
125 		"Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec",
126 		"January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"
127 	]
128 };
129 
130 // For convenience...
131 Date.prototype.format = function (mask, utc) 
132 {
133 	return dateFormat(this, mask, utc);
134 };
