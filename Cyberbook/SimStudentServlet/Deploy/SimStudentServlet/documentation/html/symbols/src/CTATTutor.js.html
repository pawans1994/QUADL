 47 /**
 48  * 
 49  * @param ctx
 50  * @param x
 51  * @param y
 52  * @param width
 53  * @param height
 54  * @param radius
 55  */
 56 function drawRoundedRect (x,y,width,height,radius)
 57 {
 58 	//debug ("drawRoundedRect ()");
 59 			
 60 	//ctx.fillStyle="#000000";
 61 	
 62 	ctx.beginPath();
 63 	ctx.moveTo(x,y+radius);
 64 	ctx.lineTo(x,y+height-radius);
 65 	ctx.quadraticCurveTo(x,y+height,x+radius,y+height);
 66 	ctx.lineTo(x+width-radius,y+height);
 67 	ctx.quadraticCurveTo(x+width,y+height,x+width,y+height-radius);
 68 	ctx.lineTo(x+width,y+radius);
 69 	ctx.quadraticCurveTo(x+width,y,x+width-radius,y);
 70 	ctx.lineTo(x+radius,y);
 71 	ctx.quadraticCurveTo(x,y,x,y+radius);
 72 	ctx.stroke();
 73 	
 74 	ctx.fillStyle="#eeeeee";
 75 	
 76 	ctx.beginPath();
 77 	ctx.moveTo(x,y+radius);
 78 	ctx.lineTo(x,y+height-radius);
 79 	ctx.quadraticCurveTo(x,y+height,x+radius,y+height);
 80 	ctx.lineTo(x+width-radius,y+height);
 81 	ctx.quadraticCurveTo(x+width,y+height,x+width,y+height-radius);
 82 	ctx.lineTo(x+width,y+radius);
 83 	ctx.quadraticCurveTo(x+width,y,x+width-radius,y);
 84 	ctx.lineTo(x+radius,y);
 85 	ctx.quadraticCurveTo(x,y,x,y+radius);	
 86 	ctx.fill();		
 87 }

406 function runTutor (flashVars)
407 {
408 	var getDebugger = new CTATBase("tutor", "");
409 	getDebugger.debug ("runTutor ("+Object.size(flashVars)+")");
410 	
411 	linkData=new CTATLinkData ();
412 	
413 	window.onerror = function(errorMsg, url, lineNumber) 
414 	{
415 	    var getDebugger = new CTATBase("", "");
416 	    getDebugger.debug("Uncaught error " + errorMsg + " in " + url + ", line " + lineNumber);
417 	};
418 	
419 	ctatcanvas = document.getElementById("canvas");
420 		
421 	if (ctatcanvas==null)
422 	{
423 		alert ("Internal error: HTML5 canvas is null");
424 		return;
425 	}
426 	
427 	ctx = ctatcanvas.getContext('2d');
428 	
429 	if (ctx==null)
430 	{
431 		alert ("Internal error: HTML5 canvas context is null");
432 		return;
433 	}	
434 	
435 	getDebugger = null;	
436 	initialize(flashVars);	
437 }
