 42 	function getNodeTextValue (aNode)
 43 	{
 44 		if (aNode==null)
 45 			return ("");
 46 		
 47 		if (aNode.childNodes==null)
 48 			return ("");
 49 		
 50 		if (aNode.childNodes.length==0)
 51 			return ("");
 52 		
 53 		return (aNode.childNodes[0].nodeValue);
 54 	}

 60 	function processMessage(aMessage,aLinkData)
 61 	{
 62 		pointer.debug ("processMessage ()");
 63 		
 64 		if (aLinkData==null)
 65 		{
 66 			pointer.debug ("Error: linkData is null");
 67 			return;
 68 		}
 69 		
 70 		linkData=aLinkData;
 71 		
 72 		var begin=getTimeStamp ();
 73 				
 74 		var xmlDoc = $.parseXML(aMessage);
 75 				 
 76 		$xml=$(xmlDoc);
 77 		
 78 		var end=getTimeStamp ();
 79 				
 80 		pointer.debug ("XML Parsing took: " + (end-begin) + " milliseconds");
 81 		
 82 		begin=getTimeStamp ();
 83 		
 84 		parseElement (xmlDoc.documentElement);
 85 		
 86 		end=getTimeStamp ();
 87 		
 88 		pointer.debug ("Start state processing took: " + (end-begin) + " milliseconds");		
 89 	}

 94 	function parseElement (anElement)
 95 	{
 96 		//debug ("parseElement ("+anElement.nodeName+")");
 97 		
 98 		if (anElement.nodeName=="StartStateMessages")
 99 			processMessageBundle (anElement);

101 		if (anElement.nodeName=="message")
102 			processSingleMessage (anElement);
103 	}

109 	{
110 		//debug ("processMessageBundle ("+anElement.nodeName+")");
111 	
112 		x=anElement.childNodes;
113 		
114 		for (var i=0;i<x.length;i++)

122 	this.processSingleMessage=processSingleMessage;
123 	function processSingleMessage (anElement)
124 	{
125 		//debug ("processSingleMessage ("+anElement.nodeName+")");
126 		
127 		var x=anElement.childNodes;
128 		
129 		for (var i=0;i<x.length;i++)
130 		{
131 			var tempElement=x [i];
132 			
133 			if (tempElement.nodeName=='properties')
134 			{
135 				//debug ("Parsing properties ...");
136 				
137 				var messageProperties=tempElement.childNodes;
138 				
139 				for (var t=0;t<messageProperties.length;t++)
140 				{
141 					var propNode=messageProperties [t];
142 															
143 					if (propNode.nodeName=="MessageType")
144 					{
145 						var nodeValue=getNodeTextValue (propNode);
146 						
147 						//debug ("MessageType: " + nodeValue);
148 						
149 						if (nodeValue=="StateGraph")
150 							processStateGraph (messageProperties);
151 					
152 						if (nodeValue=="StartProblem")
153 							processStartProblem (messageProperties);
154 					
155 						if (nodeValue=="InterfaceIdentification")
156 							processInterfaceIdentification (messageProperties);
157 					
158 						if (nodeValue=="InterfaceDescription")
159 							processInterfaceDescription (messageProperties);			
160 					
161 						if (nodeValue=="SendWidgetLock")
162 						{
163 							pointer.debug ("Found: SendWidgetLock");
164 						}	
165 					
166 						if (nodeValue=="StartStateEnd")
167 						{
168 							pointer.debug ("Found: StartStateEnd");
169 							
170 							//displayComponentList ();
171 							
172 							if (messageHandler!=null)
173 							{
174 								messageHandler.processStartState ();
175 							}	
176 						}
177 					}	
178 				}
179 			}
180 		}				
181 	}

209 	this.processInterfaceDescription=processInterfaceDescription;
210 	function processInterfaceDescription (aPropertyList)
211 	{
212 		//debug ("processInterfaceDescription ("+aPropertyList.length+")");
213 				
214 		var aType="Unknown";
215 		var aName="Unknown";
216 		
217 		for (var i=0;i<aPropertyList.length;i++)
218 		{
219 			var tempElement=aPropertyList [i];
220 						
221 			if (tempElement.nodeName=="WidgetType")
222 			{
223 				//debug ("Widget type: " + getNodeTextValue (tempElement));
224 				
225 				aType=getNodeTextValue (tempElement);
226 			}
227 			
228 			if (tempElement.nodeName=="CommName")
229 			{
230 				//debug ("Instance name: " + getNodeTextValue (tempElement));
231 				
232 				aName=getNodeTextValue (tempElement);
233 			}
234 			
235 			if (tempElement.nodeName=="serialized")
236 			{
237 				//debug ("De-serializing component ...");
238 				
239 				deserializeComponent (aType,aName,tempElement.childNodes[0]);
240 			}			
241 		}	
242 	}

246 	this.deserializeComponent=deserializeComponent;
247 	function deserializeComponent (aType,aName,aComponentElement)
248 	{
249 		//debug ("deserializeComponent ("+aType+","+aName+")");
250 		
251 		var x=aComponentElement.attributes.getNamedItem("x").value;
252 		var y=aComponentElement.attributes.getNamedItem("y").value;
253 		var width=aComponentElement.attributes.getNamedItem("width").value;
254 		var height=aComponentElement.attributes.getNamedItem("height").value;
255 		
256 		//debug ("x: " + x + ", y: " + y + ", width: " + width + ", height: " + height);
257 		
258 		var compEntry=new CTATComponentDescription ();
259 		compEntry.type=aType;
260 		compEntry.name=aName;
261 		compEntry.x=Math.round (x);
262 		compEntry.y=Math.round (y);
263 		compEntry.width=Math.round (width);
264 		compEntry.height=Math.round (height);
265 		
266 		linkData.components.push (compEntry);
267 		
268 		var serializedProps=aComponentElement.childNodes;
269 		
270 		for (var i=0;i<serializedProps.length;i++)
271 		{
272 			var tempElement=serializedProps [i];
273 						
274 			if (tempElement.nodeName=="Parameters")
275 			{
276 				//debug ("Processing component parameters ...");
277 			}
278 			
279 			if (tempElement.nodeName=="Styles")
280 			{
281 				//debug ("Processing component styles ...");
282 				
283 				var stylesProps=tempElement.childNodes [0].childNodes;
284 				
285 				for (var t=0;t<stylesProps.length;t++)
286 				{
287 					var styleProperty=stylesProps [t];
288 					
289 					if (styleProperty.nodeName=="CTATStyleProperty")
290 					{
291 						//debug ("Processing style property ...");
292 						
293 						//alert(xml_to_string(styleProperty));
294 						
295 						var aStyle=new CTATStyle ();
296 						
297 						var styleValues=styleProperty.childNodes;
298 						
299 						for (var j=0;j<styleValues.length;j++)
300 						{
301 							var styleElement=styleValues [j];
302 							
303 							//alert(xml_to_string(styleElement));
304 							
305 							if (styleElement.nodeName=="name")
306 							{
307 								//debug ("Found style name: " + getNodeTextValue (styleElement));
308 								aStyle.styleName=getNodeTextValue (styleElement);								
309 							}
310 							
311 							if (styleElement.nodeName=="value")
312 							{	
313 								//debug ("Found style value: " + getNodeTextValue (styleElement));
314 								aStyle.styleValue=getNodeTextValue (styleElement);
315 							}	
316 						}
317 																		
318 						//debug ("Style name: " + aStyle.styleName + ", value: " + aStyle.styleValue);
319 												
320 						compEntry.styles.push(aStyle);						
321 					}
322 				}	
323 			}			
324 		}	
325 	}

346     function xml_to_string(xml_node)
347     {
348         if (xml_node.xml)
349             return xml_node.xml;
350         else if (XMLSerializer)
351         {
352             var xml_serializer = new XMLSerializer();
353             return xml_serializer.serializeToString(xml_node);
354         }
355         else
356         {
357             alert("ERROR: Extremely old browser");
358             return "";
359         }
