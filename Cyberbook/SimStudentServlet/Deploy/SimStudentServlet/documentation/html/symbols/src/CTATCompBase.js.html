358 	this.getComponentFromID = function getComponentFromID (anID)
359 	{
360 		pointer.debug ("getComponentFromID ("+anID+") -> " + componentReferences.length + " references");
361 				
362 		for (var i=0;i<componentReferences.length;i++)
363 		{
364 			var ref=componentReferences [i];
365 						
366 			if (ref.getDiv ().getAttribute ("id")==anID)
367 			{				
368 				if (ref.getComponent ()==null)
369 				{
370 					pointer.debug ("Error: found component reference is null! ("+ref.componentReference+")");
371 					return (null);
372 				}
373 				
374 				pointer.debug ("Found component reference: " + ref.getComponent ().getName () + " for div: " + anID);
375 											
376 				return (ref.getComponent ());
377 			}
378 		}
379 		
380 		pointer.debug ("Bottoming out ...");
381 		
382 		return (null);
383 	}

641 	this.processKeypress=function processKeypress (e)
642 	{
643 		pointer.debug ("processKeypress ()");
644 		
645 		var id=e.currentTarget.getAttribute ("id");
646 		var comp=pointer.getComponentFromID (id);
647 		
648 		if (comp==null)
649 		{
650 			pointer.debug ("Error: component reference is null");
651 			return;
652 		}
653 		
654 		pointer.debug (comp.name + " keydown ("+getKey (e)+" -> "+e.eventPhase+") " + "ID: " + id);
655 								
656 		currentComponent=id;
657 		currentComponentPointer=comp;
658 		
659 	    switch (getKey (e)) 
660 	    {
661 	    	// key code for left arrow
662 	       	case 37:
663 	       			pointer.debug('left arrow key pressed!');
664 	       			break;
665 		        
666 	       			// key code for right arrow
667 	       	case 39:
668 	       			pointer.debug('right arrow key pressed!');
669 	       			break;
670 	       			
671 	       	case 13:
672 	       			pointer.debug('Enter key pressed!');
673 	       			//commShell.focusNextComponent (comp);
674 	       			commShell.gradeComponent (comp);
675 	       			return (false);
676 	    }		
677 	};

679 	/**
680 	 * This method is called when a component gets focused. So keep in mind that
681 	 * we want to grade the component that lost focus, or in other words the
682 	 * previously focused component.
683 	 */
684 	this.processFocus=function processFocus (e)
685 	{
686 		pointer.debug ("processFocus ()");
687 		
688 		var id=e.currentTarget.getAttribute ("id");
689 		var comp=pointer.getComponentFromID (id);
690 		
691 		if (oldComponentFocus!=null)
692 		{
693 			if (oldComponentFocus==comp)
694 			{
695 				pointer.debug ("We're already there!");
696 				return;
697 			}
698 			
699 			commShell.gradeComponent (oldComponentFocus);
700 			oldComponentFocus.highlightComponent(false);
701 		}
702 		else
703 			pointer.debug ("No previously focused component yet, can't grade");
704 		
705 		if (comp==null)
706 		{
707 			pointer.debug ("Error: component reference is null");
708 			return;
709 		}			
710 					
711 		pointer.highlightComponent(true);
712 		
713 		oldComponentFocus=pointer;			
714 	};
