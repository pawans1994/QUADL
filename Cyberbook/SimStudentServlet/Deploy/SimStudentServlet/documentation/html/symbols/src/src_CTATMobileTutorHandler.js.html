 20 */
 21 var stringInput = "";
 22 var swfObjName="SWF";
 23 var mode="disabled";
 24 
 25 function CTATMobileTutorHandler(aName,aMode)
 26 {
 27 	CTATBase.call(this, "CTATMobileTutorHandler","mobiletutorhandler");
 28 	
 29 	this.debug ("CTATMobileTutorHandler ()");
 30 	this.debug ("Edit moved sI");
 31 	
 32 	var swfObjName=aName;
 33 	var mode=aMode;
 34 	var onDevice=false;
 35 	
 36 	if( /Android|webOS|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini/i.test(navigator.userAgent) ) 
 37 	{
 38 		onDevice=true;
 39 	}
 40 	
 41 	/**
 42 	*
 43 	*/	
 44 	this.error=function error (aMessage)
 45 	{
 46 		this.debug (aMessage);
 47 		
 48 		alert (aMessage);
 49 	}
 50 		
 51 	/**
 52 	 * 
 53 	 * @param aComponent One of 'landscape' or 'portrait'
 54 	 */
 55 	this.processOrientationChange=function processOrientationChange (orientation)
 56 	{
 57 		this.debug ("processOrientationChange ("+orientation+")");
 58 		
 59 	}	
 60 	
 61 	/**
 62 	 * The argument will be a component with a className of either one of:
 63 	 * CTATTextInput, CTATTextField, CTATTextArea
 64 	 * 
 65 	 * @param aComponent
 66 	 */
 67 	this.processTextFocus=function processTextFocus (x,y,width,height)
 68 	{
 69 		this.debug ("processTextFocus ("+x+","+y+","+width+","+height+")");
 70 
 71 		if (mode=="disabled")
 72 		{
 73 			return;
 74 		}
 75 
 76 		if ((mode=="auto") && (onDevice==false))
 77 		{
 78 			return;
 79 		}
 80 		
 81 		stringInput = "";
 82 		
 83 		this.manipulateKeyboard();
 84 
 85 		//this.setText ("Hello: " + Math.random());
 86 	}	
 87 	
 88 	/**
 89 	*
 90 	*/
 91 	this.processEnter=function processEnter ()
 92 	{
 93 		this.debug ("processEnter ()");
 94 		
 95 		if (mode=="disabled")
 96 		{
 97 			return;
 98 		}
 99 		
100 		if ((mode=="auto") && (onDevice==false))
101 		{
102 			return;
103 		}		
104 		
105 		var swfObject=document.getElementById (swfObjName);			
106 			
107 		if (swfObject!=null)
108 		{												
109 			try
110 			{
111 				swfObject.processExternalEnter ();
112 
113 				this.debug ("Successfully called AS3 method");
114 			}
115 			catch(err)
116 			{
117 				this.debug ("Error description: " + err.message);
118 			}					
119 		}
120 		else
121 			this.debug ("Error: unable to obtain reference to swf object");							
122 	}
123 	
124 	/**
125 	 * Called by any outside code to push a new string into the currently
126 	 * selected text input component
127 	 */
128 	this.setText=function setText (aString)
129 	{	
130 		this.debug ("setText ("+aString+")");
131 		
132 		if (mode=="disabled")
133 		{
134 			return;
135 		}
136 		
137 		if ((mode=="auto") && (onDevice==false))
138 		{
139 			return;
140 		}		
141 		
142 		if (selectedTextInput!=null)
143 		{
144 			this.debug ("Attempting to call HTML5 method on text object ...");
145 			
146 			selectedTextInput.setText (aString);
147 		}
148 		else
149 		{	
150 			this.debug ("Attempting to call AS3 method ...");
151 			
152 			var swfObject=document.getElementById(swfObjName);
153 			//this.debug(swfObject);
154 			
155 			if (swfObject!=null)
156 			{
157 				//this.debugObject (swfObject);
158 												
159 				try
160 				{
161 					stringInput = stringInput + aString;
162 					this.debug (stringInput);
163 					swfObject.processExternalKeyboard(stringInput);
164 
165 					this.debug ("Successfully called AS3 method");
166 				}
167 				catch(err)
168 				{
169 					this.debug ("Error description: " + err.message);
170 				}					
171 			}
172 			else
173 				this.debug ("Error: unable to obtain reference to swf object");										
174 		}
175 	}
176 	
177 	/**
178 	*
179 	*/
180 	this.manipulateKeyboard=function manipulateKeyboard()
181 	{		
182 		this.debug ("manipulateKeyboard()");
183 		
184 		if (mode=="disabled")
185 		{
186 			return;
187 		}
188 		
189 		if ((mode=="auto") && (onDevice==false))
190 		{
191 			return;
192 		}		
193 		
194 		document.getElementById("keyboardUI").style.visibility="visible";
195 		
196 		
197 	};
198 }
