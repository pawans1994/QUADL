 96 	{
 97 		this.debug ("gradeComponent ()");
 98 			
 99 		if (aComponent==null)
100 		{
101 			this.debug ("Internal error, provided component is null");
102 			return;
103 		}
104 		
105 		this.debug ("Grading: " + aComponent.getName ());
106 		
107 		if ((aComponent.getClassName ()=="CTATTextInput") || (aComponent.getClassName ()=="CTATTextField") || (aComponent.getClassName ()=="CTATTextArea"))
108 		{
109 			this.debug ("Backgrading ...");
110 			
111 			this.debug ("Grading value: " + aComponent.getValue ());
112 			
113 			if (isBlank (aComponent.getValue ())==true)
114 			{
115 				this.debug ("Empty component, nothing to grade");
116 				return;
117 			}
118 			
119 			aComponent.showCorrect();
120 			
121 			return;
122 		}
123 
124 		this.debug ("Grading regular component ...");
125 						
126 		if (aComponent.getName ()=="done")
127 		{
128 			// kick in done
129 			
130 			return;
131 		}
132 		
133 		if (aComponent.getName ()=="hint")
134 		{

145 			
146 			return;
147 		}
148 		
149 		this.debug ("Component value: " + aComponent.getValue ());
150 		
151 		this.debugObjectShallow (aComponent.getComponent ());
