 16 The scrim. Modeled after the scrim in ActionScript 3.0.
 17 
 18 At the moment, a global variable called ctatscrim (CTATGlobals.js) should be used as the scrim.
 19 
 20 Here are is a list of the different methods:
 21 	- scrimUp(aMessage) : Pulls up a scrim with a dialog that displays a message aMessage.
 22 	- waitScrimUp() : Pulls up the wait scrim.
 23 	- okScrimUp(aMessage, aFunction) : Pulls up a scrim with a dialog message of aMessage, and executes aFunction when the ok button is clicked.
 24 	- confirmScrimUp(prompt, onYes, onNo) : Pulls up a scrim with a prompt message, and function handlers for clicking on yes/no buttons.
 25 	- errorScrimUp(aMessage) : Pulls up a non-removable scrim displaying an error message aMessage.
 26 	- warningScrimUp(aMessage) : Pulls up scrim displaying a warning message aMessage, and may only be closed when the close button is clicked.
 27 	- scrimDown() : Pulls down the scrim if it is in a state that allows it to be pulled down.
 28 	- resizeScrim(newWidth, newHeight) : Resizes the scrim, and sets the width and height to newWidth and newHeight.
 29 
 30 If the wait scrim is up, and you wish to pull up another scrim, you should pull down the wait scrim first for the time
 31 being.
 32 
 33 */

118 	/**
119 	 * Inspects the input dialog text to see if it flows over the
120 	 * dialog box or not. If it does, then the message is split
121 	 * up into multiple messages that will instead be placed from
122 	 * top to bottom. This must be called before fitTextToDialogY.
123 	 */
124 	function fitTextToDialogX(dialogText, rightMargin, leftMargin)
125 	{
126 		var separatedMessage="";
127 		
128 		/**
129 		 * Converts font points to pixels. No *real* reason why this equation
130 		 * is the way it is, but it seems to work ok. Only an estimate.
131 		 */
132 		var fontPoint=scrimMessage[0].getTextSize();
133 		estPointToPixel=((fontPoint / 2) + 1);
134 		
135 		var textLength = leftMargin+(dialogText.length * estPointToPixel);
136 		
137 		if(textLength > rightMargin)
138 		{
139 			var textArray=dialogText.split(" ");
140 			
141 			for(var i=0; i<textArray.length;i++)
142 			{
143 				if(textArray[i]=="")
144 				{
145 					textArray.splice(i, (textArray.length-i));
146 				}
147 			}
148 			
149 			//If the dialogText.split() call returns one element, it is because what is left is
150 			//just one long token. We need to handle this case differently.
151 			if(textArray.length==1)
152 			{
153 				var t=textArray[0];
154 				
155 				for(var i=0; i < t.length; i++)
156 				{
157 					textArray[i]=t[i];
158 				}
159 			}
160 			
161 			
162 			//Find out where we get cut off, then say the new message is everything
163 			//after the cutoff point. Call this function again to check if the
164 			//new message will be able to fit in the dialog box or not.
165 			
166 			var textSizeSum=leftMargin;
167 			var temp="";
168 
169 			for(var i=0; i < textArray.length; i++)
170 			{
171 				textSizeSum += (textArray[i].length * estPointToPixel);
172 
173 				if((textSizeSum > rightMargin) || (textArray[i]=="\n"))
174 				{
175 					//Do not count the newline char, or you will get stack overflow
176 					if(textArray[i]=="\n") 
177 					{
178 						i++;
179 					}
180 					
181 					for(var j=i; j < textArray.length; j++)
182 					{
183 						separatedMessage += textArray[j] + " ";
184 					}
185 
186 					textArray.splice(i, textArray.length);
187 					fitTextToDialogX(separatedMessage, rightMargin, leftMargin);
188 				}
189 
190 				else
191 				{
192 					temp += (textArray[i] + " ");
193 				}
194 			}
195 
196 			tempList.push(temp);
197 		}
198 		else
199 		{
200 			tempList.push(dialogText);
201 		}
202 	}

204 	/**
205 	 * Stretches the dialog box to fit the entire message, if necessary.
206 	 * Must call fitTextToDialogX first. Returns the new height.
207 	 */
208 	function fitTextToDialogY(y, botMargin, marginSize)
209 	{
210 		//How much space the text uses up
211 		var textSize=y+((estPointToPixel+padding) * (messageList.length+1));
212 		
213 		if(textSize > botMargin)
214 		{
215 			return (textSize+marginSize);
216 		}
217 		
218 		return (dialog.getHeight()+marginSize);
219 	}

221 	function displayDialog(aX, aY, aWidth, aHeight)
222 	{		
223 		/**
224 		 * This should really be left alone. It seems that changing it messes up
225 		 * the dialog box. If it is crucial that the margin size be changed, it
226 		 * is probably necessary to re-write this function to be margin-sensitive.
227 		 */
228 		var marginRatio=1/8;
229 		
230 		var dialogLeftMargin=(aX+(aWidth * marginRatio));
231 		var dialogRightMargin=(aX+(aWidth-(aWidth * marginRatio)));
232 		
233 		var yMarginSize=(aHeight * marginRatio);
234 		
235 		var dialogTopMargin=(aY+yMarginSize);
236 		var dialogBottomMargin=(aY+(aHeight-yMarginSize));
237 
238 		dialog=new CTATShape("container", 
239 							 "scrim dialog",
240 							 aX,
241 							 aY,
242 							 aWidth,
243 							 aHeight);
244 							 
245 		dialog.addPoint(0, 0);
246 
247 		dialog.modifyCanvasCSS("z-index", "998");
248 
249 		scrimMessage=new Array();
250     
251     	scrimComponent.addShape(dialog);
252     	
253        /**
254     	* We need the text size to do text wrapping. Since the text uses the same font
255     	* for all messages, we can just use a dummy object to extract this.
256 		*/
257 		scrimMessage[0]=new CTATShape("container", " ", [0], [0], 0, 0);
258 		 
259 		fitTextToDialogX(message, dialogRightMargin, dialogLeftMargin);
260 		
261 		scrimMessage[0].detatchCanvas();
262 
263 		for(var i=tempList.length-1; i >= 0; i--)
264 		{
265 			messageList.push(tempList[i]);
266 		}
267 		
268 		dialog.setHeight(fitTextToDialogY(aY, dialogBottomMargin, yMarginSize));
269 
270 		//If the text was bigger than the box, then the dialog box will need re-centered
271 		var mainCanvas=document.getElementById("main-canvas");
272 		var newY=((mainCanvas.height/2)-(dialog.getHeight()/2));
273 		scrimComponent.moveShape(dialog.getName(), dialog.getXOffset(), newY);
274 
275 		dialogTopMargin=(newY+(dialog.getHeight() * marginRatio));
276 		
277 		dialog.setLineColor(scrimBorderColor);
278 		dialog.setFillColor("white");
279 		dialog.setDrawWidth(scrimBorderWidth);
280 		dialog.setRadius(5);
281 		dialog.drawRoundedRectFilled();
282 
283 		for(var i=0; i < messageList.length; i++)
284 		{
285 			var y=(dialogTopMargin+((estPointToPixel+padding)*(i+1)));
286 		
287 			scrimMessage[i]=new CTATShape("container", 
288 							 	   		  ("scrim message "+i),
289 							 	  		  dialogLeftMargin,
290 							       		  y,
291 							       		  aWidth,
292 							       		  aHeight);
293 							       		  
294 			scrimMessage[i].addPoint(0, 0);
295 		
296 			scrimMessage[i].modifyCanvasCSS("z-index", "999");       
297 			scrimMessage[i].setColor("black");
298 			scrimMessage[i].setTextFont("Verdana");
299 			scrimMessage[i].setTextSize(14);
300 			scrimMessage[i].setData(messageList[i]);
301 			scrimMessage[i].drawText();
302 			scrimComponent.addShape(scrimMessage[i]);
303 		}
304 							 
305 		if(hasCloseButton==true)
306 		{
307 			var buttonSizeY=20;
308 			var buttonSizeX=40;
309 
310 			var lastMessage=scrimMessage[messageList.length-1];
311 			var spaceUsedUp=lastMessage.getYOffset()+lastMessage.getTextSize()+padding;
312 			
313 			if((spaceUsedUp + buttonSizeY) > dialogBottomMargin) 
314 			{
315 				yMarginSize=((spaceUsedUp+buttonSizeY) * marginRatio);
316 				dialog.setHeight(spaceUsedUp+buttonSizeY+yMarginSize+scrimBorderWidth);
317 				var placement = ((mainCanvas.height/2) - (dialog.getHeight()/2));
318 				
319 				scrimComponent.moveShape(dialog.getName(), dialog.getXOffset(), placement);
320 				
321 				dialog.setLineColor(scrimBorderColor);
322 				dialog.setFillColor("white");
323 				dialog.setDrawWidth(scrimBorderWidth);
324 				dialog.setRadius(5);
325 				dialog.drawRoundedRectFilled();
326 			}
327 			
328 			closeButton=makeHTMLButton(closeFunction, (aX+(dialog.getWidth()/2)-(buttonSizeX/2)), 
329 			                           spaceUsedUp, "close");
330 		}
331 		
332 		if(hasYesButton==true)
333 		{
334 			var buttonSizeY=20;
335 			var buttonSizeX=40;
336 
337 			var lastMessage=scrimMessage[messageList.length-1];
338 			var spaceUsedUp=lastMessage.getYOffset()+lastMessage.getTextSize()+padding;
339 			
340 			if((spaceUsedUp + buttonSizeY) > dialogBottomMargin) 
341 			{
342 				yMarginSize=((spaceUsedUp+buttonSizeY) * marginRatio);
343 				dialog.setHeight(spaceUsedUp+buttonSizeY+yMarginSize+scrimBorderWidth);
344 				var placement = ((mainCanvas.height/2) - (dialog.getHeight()/2));
345 				
346 				scrimComponent.moveShape(dialog.getName(), dialog.getXOffset(), placement);
347 				
348 				dialog.setLineColor(scrimBorderColor);
349 				dialog.setFillColor("white");
350 				dialog.setDrawWidth(scrimBorderWidth);
351 				dialog.setRadius(5);
352 				dialog.drawRoundedRectFilled();
353 			}
354 			
355 			yesButton=makeHTMLButton(yesPtr, (aX+(dialog.getWidth()/2)-buttonSizeX), 
356 			                         spaceUsedUp, "yes");
357 		}
358 		
359 		if(hasNoButton==true)
360 		{
361 			var buttonSizeY=20;
362 			var buttonSizeX=40;
363 
364 			var lastMessage=scrimMessage[messageList.length-1];
365 			var spaceUsedUp=lastMessage.getYOffset()+lastMessage.getTextSize()+padding;
366 			
367 			if((spaceUsedUp + buttonSizeY) > dialogBottomMargin) 
368 			{
369 				yMarginSize=((spaceUsedUp+buttonSizeY) * marginRatio);
370 				dialog.setHeight(spaceUsedUp+buttonSizeY+yMarginSize+scrimBorderWidth);
371 				var placement = ((mainCanvas.height/2) - (dialog.getHeight()/2));
372 				
373 				scrimComponent.moveShape(dialog.getName(), dialog.getXOffset(), placement);
374 				
375 				dialog.setLineColor(scrimBorderColor);
376 				dialog.setFillColor("white");
377 				dialog.setDrawWidth(scrimBorderWidth);
378 				dialog.setRadius(5);
379 				dialog.drawRoundedRectFilled();
380 			}
381 			
382 			noButton=makeHTMLButton(noPtr, (aX+(dialog.getWidth()/2)), 
383 			                        spaceUsedUp, "no");
384 		}
385 		
386 		if(waitScrim==true)
387 		{
388 			var spaceUsedUp=(newY+((messageList.length+1) * (estPointToPixel + padding)));
389 		
390 			//The gif we use is 64x64
391 			var waitGIFsize=64;
392 			var gifMid=(aX+(dialog.getWidth()/2)-(waitGIFsize/2));
393 			
394 			//If the gif is too big, re-size and re-position
395 			if((spaceUsedUp+waitGIFsize) > dialogBottomMargin)
396 			{
397 				yMarginSize=((spaceUsedUp+waitGIFsize) * marginRatio);
398 				dialog.setHeight(spaceUsedUp+waitGIFsize+yMarginSize);
399 				var placement = ((mainCanvas.height/2) - (dialog.getHeight()/2));
400 				
401 				scrimComponent.moveShape(dialog.getName(), dialog.getXOffset(), placement);
402 				
403 				dialog.setLineColor(scrimBorderColor);
404 				dialog.setFillColor("white");
405 				dialog.setRadius(5);
406 				dialog.drawRoundedRectFilled();
407 			}
408 		
409 			makeHTMLImage(rotatingLoad, gifMid, (spaceUsedUp+dialog.getTextSize()+padding));
410 		}

444 	{
445 		pointer.debug("scrimUp ()");
446 		pointer.debug("Message: " + aMessage);
447 
448 		if(scrimIsUp==true)
449 		{
450 			if(waitScrim==true)
451 			{
452 				message=aMessage;
453 			}
454 			
455 			else
456 			{
457 				message += " \n " + aMessage;
458 			}
459 			
460 			//Only need one scrim up at a time
461 			scrimComponent.removeComponent();
462 			messageList=new Array();
463 			tempList=new Array();
464 			removeHTMLElements();
465 		}
466 		
467 		else
468 		{
469 			message=aMessage;
470 			scrimIsUp=true;
471 		}
472 		
473 		drawScrim();
474 	};

645 	{
646 		if (scrimIsUp==false) 
647 		{
648 			pointer.debug("The scrim isn't up, returning");
649 			return;
650 		}
651 			
652 		if (errorScrim==true) 
653 		{
654 			pointer.debug("The scrim is up to describe errors to the user, leave it up!");
655 			return;
656 		}
657 			
658 		if (warnScrim==true) 
659 		{
660 			pointer.debug("The scrim is up to display warnings to the user, leave it up!");
661 			return;
662 		}
663 			
664 		if (connectionScrim==true) 
665 		{
666 			pointer.debug("The scrim is up for a connection issue leave it up!");
667 			return;
668 		}
669 		/*
670 		if (waitForClick) 
671 		{
672 			pointer.debug("The scrim is waiting on user input, leave it up!");
673 			return;
674 		}*/
675 		
676 		scrimComponent.removeComponent();
677 		removeHTMLElements();
678 		scrimIsUp=false;
679 		waitScrim=false;
680 		message="";
