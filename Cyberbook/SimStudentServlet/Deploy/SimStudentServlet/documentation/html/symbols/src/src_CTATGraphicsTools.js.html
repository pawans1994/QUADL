152 	/**------------------------------------------------------------------------------------
153 	* drawLine(x1,y1,x2,y2,c,w):void
154 	* @param x1: the x-coordinate of the first point
155 	* @param y1: the y-coordinate of the first point
156 	* @param x2: the x-coordinate of the second point
157 	* @param y2: the y-coordinate of the second point
158 	* @param c: color of the line (May not be correct comment)
159 	* @param w: width of the line (May not be correct comment)
160 	* Draws a line
161 	**------------------------------------------------------------------------------------*/
162 	this.drawLine=function drawLine(x1, y1, x2, y2, aColor, aWidth)
163 	{
164 		if(canvasCtx==null)
165 		{
166 			pointer.debug("Error: canvas canvasCtx is NULL in drawLine");
167 		}	
168 		
169 		var finalX1=x1+aliasingOffset;
170 		var finalX2=x2+aliasingOffset;
171 		var finalY1=y1+aliasingOffset;
172 		var finalY2=y2+aliasingOffset;	
173 		
174 		canvasCtx.lineWidth=aWidth;
175 		canvasCtx.strokeStyle=aColor;
176 		
177 		canvasCtx.beginPath();
178 		canvasCtx.moveTo(finalX1, finalY1);
179 		canvasCtx.lineTo(finalX2, finalY2);
180 		canvasCtx.stroke();
181 	};

183 	/**------------------------------------------------------------------------------------
184 	* drawCircle(centerX, centerY, radius, sides):void
185 	* @param centerX: the x-coordinate of the center
186 	* @param centerY: the y-coordinate of the center
187 	* @param radius: the radius of the circle
188 	* @param sides: the number of sides
189 	* NOTE: Due to restrictions, drawCircle is drawing an n-gon with 'sides' sides
190 	* Draws a circle
191 	**------------------------------------------------------------------------------------*/
192 	this.drawCircle=function drawCircle(x, y, radius)
193 	{
194 		if(canvasCtx==null)
195 		{
196 			pointer.debug("Error: canvas canvasCtx is NULL in drawCircle");
197 		}
198 		
199 		var finalX=x+aliasingOffset;
200 		var finalY=y+aliasingOffset;
201 	
202 		canvasCtx.strokeStyle=lineColor;
203 		canvasCtx.lineWidth=borderWidth;
204 		
205 		canvasCtx.beginPath();
206 		canvasCtx.arc(finalX, finalY, radius, 0, 2*Math.PI);
207 		canvasCtx.stroke();
208 	};

210 	/**------------------------------------------------------------------------------------
211 	* drawCircleFilled(centerX:Number, centerY:Number, radius:Number, sides:Number):void

213 	* @param centerY: the y-coordinate of the center
214 	* @param radius: the radius of the circle
215 	* @param sides: the number of sides
216 	* NOTE: Due to restrictions, drawCircle is drawing an n-gon with 'sides' sides
217 	* Draws a filled circle
218 	**------------------------------------------------------------------------------------*/
219 	this.drawCircleFilled=function drawCircleFilled(x, y, radius)
220 	{

222 		{

224 		}
225 		
226 		var finalX=x+aliasingOffset;
227 		var finalY=y+aliasingOffset;
228 		
229 		canvasCtx.fillStyle=fillColor;
230 		
231 		canvasCtx.beginPath();
232 		canvasCtx.arc(finalX, finalY, radius, 0, 2*Math.PI);
233 		canvasCtx.fill();
234 	};

236 	/**------------------------------------------------------------------------------------
237 	* drawRectangle (x:Number,y:Number,w:Number,h:Number)
238 	* @param x: the x-coordinate of the top left corner of the rectangle
239 	* @param y: the y-coordinate of the top left corner of the rectangle
240 	* @param w: the width of th rectangle
241 	* @param h: the height of the rectangle
242 	* Draws a rectangle
243 	**------------------------------------------------------------------------------------*/
244 	this.drawRectangle=function drawRectangle(x, y, aWidth, aHeight)
245 	{
246 		if(canvasCtx==null)
247 		{
248 			pointer.debug("Error: canvas canvasCtx is NULL in drawRectangle");
249 		}
250 	
251 		var finalX=x+aliasingOffset;
252 		var finalY=y+aliasingOffset;
253 		
254 		canvasCtx.lineWidth=borderWidth;
255 		
256 		if(fillShape==true)
257 		{
258 			canvasCtx.fillStyle=fillColor;
259 			canvasCtx.fillRect(finalX, finalY, aWidth, aHeight);
260 		}
261 		
262 		else
263 		{
264 			canvasCtx.strokeStyle=lineColor;
265 			canvasCtx.beginPath();
266 			canvasCtx.rect(finalX, finalY, aWidth, aHeight);
267 			canvasCtx.stroke();
268 		}
269 	};

297 	 * @param x

299 	 * @param width
300 	 * @param height
301 	 * @param radius
302 	 */
303 	this.drawRoundedRect=function drawRoundedRect (x,y,aWidth,aHeight,radius)
304 	{		
305 		var finalX=x+aliasingOffset;
306 		var finalY=y+aliasingOffset;
307 		
308 		//pointer.debug ("drawRoundedRect ("+finalX+","+finalY+","+aWidth+","+aHeight+","+radius+")");
309 		
310 		if (canvasCtx==null)
311 			return;
312 		
313 		canvasCtx.strokeStyle=lineColor;
314 		canvasCtx.lineWidth=borderWidth;
315 		
316 		canvasCtx.beginPath();
317 		canvasCtx.moveTo(finalX,finalY+radius);
318 		canvasCtx.lineTo(finalX,finalY+aHeight-radius);
319 		canvasCtx.quadraticCurveTo(finalX,finalY+aHeight,finalX+radius,finalY+aHeight);
320 		canvasCtx.lineTo(finalX+aWidth-radius,finalY+aHeight);
321 		canvasCtx.quadraticCurveTo(finalX+aWidth,finalY+aHeight,finalX+aWidth,finalY+aHeight-radius);
322 		canvasCtx.lineTo(finalX+aWidth,finalY+radius);
323 		canvasCtx.quadraticCurveTo(finalX+aWidth,finalY,finalX+aWidth-radius,finalY);
324 		canvasCtx.lineTo(finalX+radius,finalY);
325 		canvasCtx.quadraticCurveTo(finalX,finalY,finalX,finalY+radius);
326 		canvasCtx.stroke();
327 	};

329 	/**
330 	 * 
331 	 * @param grcanvasCtx
332 	 * @param x
333 	 * @param y
334 	 * @param width
335 	 * @param height
336 	 * @param radius
337 	 */
338 	this.drawRoundedRectFilled=function drawRoundedRectFilled (x,y,aWidth,aHeight,radius)
339 	{		
340 		var finalX=x+aliasingOffset;
341 		var finalY=y+aliasingOffset;
342 
343 		//pointer.debug ("drawRoundedRect ("+finalX+","+finalY+","+aWidth+","+aHeight+","+radius+")");
344 		
345 		if (canvasCtx==null)
346 			return;
347 		
348 		// Draw outline ...
349 		
350 		canvasCtx.strokeStyle=lineColor;
351 		canvasCtx.lineWidth=borderWidth;
352 
353 		canvasCtx.beginPath();
354 		canvasCtx.moveTo(finalX,finalY+radius);
355 		canvasCtx.lineTo(finalX,finalY+aHeight-radius);
356 		canvasCtx.quadraticCurveTo(finalX,finalY+aHeight,finalX+radius,finalY+aHeight);
357 		canvasCtx.lineTo(finalX+aWidth-radius,finalY+aHeight);
358 		canvasCtx.quadraticCurveTo(finalX+aWidth,finalY+aHeight,finalX+aWidth,finalY+aHeight-radius);
359 		canvasCtx.lineTo(finalX+aWidth,finalY+radius);
360 		canvasCtx.quadraticCurveTo(finalX+aWidth,finalY,finalX+aWidth-radius,finalY);
361 		canvasCtx.lineTo(finalX+radius,finalY);
362 		canvasCtx.quadraticCurveTo(finalX,finalY,finalX,finalY+radius);
363 		canvasCtx.stroke();
364 		
365 		//fill ...
366 		canvasCtx.fillStyle=fillColor;
367 	
368 		canvasCtx.beginPath();
369 		canvasCtx.moveTo(finalX,finalY+radius);
370 		canvasCtx.lineTo(finalX,finalY+aHeight-radius);
371 		canvasCtx.quadraticCurveTo(finalX,finalY+aHeight,finalX+radius,finalY+aHeight);
372 		canvasCtx.lineTo(finalX+aWidth-radius,finalY+aHeight);
373 		canvasCtx.quadraticCurveTo(finalX+aWidth,finalY+aHeight,finalX+aWidth,finalY+aHeight-radius);
374 		canvasCtx.lineTo(finalX+aWidth,finalY+radius);
375 		canvasCtx.quadraticCurveTo(finalX+aWidth,finalY,finalX+aWidth-radius,finalY);
376 		canvasCtx.lineTo(finalX+radius,finalY);
377 		canvasCtx.quadraticCurveTo(finalX,finalY,finalX,finalY+radius);	
378 		canvasCtx.fill();
379 	};

381 	/**
382 	 * Text is placed from a top left anchor.
383 	 * @param x
384 	 * @param y
385 	 * @param aText
386 	 */
387 	this.drawText=function drawText (x,y,aText)
388 	{				
389 		//pointer.debug ("drawText ("+anX+","+anY+","+aText+")");
390 		
391 		if (canvasCtx==null)
392 			return;		
393 		
394 		var finalX=x+aliasingOffset;
395 		var finalY=y+aliasingOffset;
396 		
397 		try
398 		{
399 			canvasCtx.fillStyle=lineColor;
400 			canvasCtx.font=(textFont+" "+textSize+"pt");
401 			canvasCtx.fillText (aText,finalX,finalY);
402 		}
403 		
404 		catch (e)
405 		{
406 			alert (incompatibleBrowserMessage);
407 			throw new Error ("Browser does not support fillText");
408 		}
409 	};
