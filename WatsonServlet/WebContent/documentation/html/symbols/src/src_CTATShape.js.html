 16  This is best used for dynamic objects. If you are simply painting something "static" (rounded rect for
 17  the Skill and Hint Windows for example), just use the CTATGraphicsTools code to paint.
 18  
 19  Use this with CTATCanvasComponent to create aggregates of shapes, or a "whole component" made
 20  of shapes. For instance, a CTATNumberLine would consist of many CTATShapes (many lines
 21  drawn), that will represent tick marks.
 22  
 23  The constructor takes six arguments:
 24  	1. anID    - ID of the div you want to add the canvas to.
 25  	2. aName   - The name of this shape.
 26  	3. startX  - The x coordinate of the canvas to be drawn on.
 27  	4. startY  - The y coordinate of the canvas to be drawn on.
 28  	5. aWidth  - The width of the canvas and the shape.
 29  	6. aHeight - The height of the canvas and the shape.
 30  	
 31 The startX and startY coordinates are absolutely positioned. Once set, you must then call
 32 addPoints which will set an (x, y) coordinate local to the shape's canvas.
 33 
 34 */
 35 
 36 function CTATShape(anID, aName, startX, startY, aWidth, aHeight)
 37 {
 38 	var pointer=this;
 39 
 40 	if((anID==undefined) || (anID==null))
 41 	{
 42 		pointer.debug("--- Error: Must pass a valid ID. ---");
 43 		return null;
 44 	}
 45 
 46 	//Holds an array of x or y coordinates to support draw methods.
 47 	var x=new Array();
 48 	var y=new Array();
 49 
 50 	var width=aWidth;
 51 	var height=aHeight;
 52 	var shapeName=aName;
 53 	
 54 	var canvasX=startX;
 55 	var canvasY=startY;
 56 	
 57 	//If you want to pair up some data with a particular shape.
 58 	//*note: this is used to hold the text for draw text
 59 	var data=undefined;
 60 	
 61 	var lineColor="black";
 62 	var fillColor="black";
 63 	
 64 	var textFont="Arial";
 65 	var textSize=12;
 66 	
 67 	var eventHandlerList=new Array();
 68 	
 69 	//This is the width of the line to be drawn. Not applicable to fill methods.
 70 	var shapeLineWidth=1;
 71 	
 72 	var radius=5;
 73 		
 74 	var canvasObject=document.createElement('canvas');
 75 	canvasObject.id=shapeName;
 76 	canvasObject.width=width;
 77 	canvasObject.height=height;
 78 		
 79 	var canvasStyles=new CTATCSS();
 80 
 81 	canvasStyles.addCSSAttribute("position", "absolute");
 82 	canvasStyles.addCSSAttribute("left", canvasX+"px");
 83 	canvasStyles.addCSSAttribute("top", canvasY+"px");
 84 		
 85 	//Lets us see what we draw on the canvas...
 86 	canvasStyles.addCSSAttribute("visibility", "visible");
 87 	
 88 	//...without making us see the canvas itself *needs IE support
 89 	canvasStyles.addCSSAttribute("background-color", "rgba(0, 0, 0, 0)");
 90 	canvasStyles.addCSSAttribute("border-style", "none");	
 91 		
 92 	//Set the style of the canvas object to the CSS attributes set thus far
 93 	canvasObject.setAttribute('style', canvasStyles.toString());
 94 	
 95 	if(anID=="container")
 96 	{
 97 		document.getElementById(anID).appendChild(canvasObject);
 98 	}
 99 	
100 	else
101 	{
102 		document.getElementById('ctatdiv'+anID).appendChild(canvasObject);
103 	}
104 
105 	var graphicTools=new CTATGraphicsTools(canvasObject.getContext("2d"));
106 	graphicTools.setGraphicBorderWidth(shapeLineWidth);
107 	graphicTools.setLineColor(lineColor);
108 	graphicTools.setFillColor(fillColor);
109 	
110 	this.addListener=function addListener(eventName, handler)
111 	{
112 		eventHandlerList.push([eventName, handler]);
113 		canvasObject.addEventListener(eventName, handler);
114 	};
115 	
116 	/**
117 	 * The get offset methods are primarily used in situations where you want to place
118 	 * one shape's canvas over another. Since a shape's coordinates are represented locally
119 	 * to its own canvas, this is needed to get the absolute positioning of its canvas.
120 	 */
121 	this.getXOffset=function getXOffset()
122 	{
123 		return (startX + x[0]);
124 	};
125 	
126 	
127 	this.getYOffset=function getYOffset()
128 	{
129 		return (startY + y[0]);
130 	};
131 	
132 	this.addPoint=function addPoint(addX, addY)
133 	{
134 		x.push(addX);
135 		y.push(addY);
136 	};
137 	
138 	this.getName=function getName()
139 	{
140 		return (shapeName);
141 	};
142 	
143 	this.getXPoint=function getXPoint(index)
144 	{
145 		return (x[index]);
146 	};
147 	
148 	this.getYPoint=function getYPoint(index)
149 	{
150 		return (y[index]);
151 	};
152 	
153 	this.getWidth=function getWidth()
154 	{
155 		return (width);
156 	};
157 	
158 	this.getHeight=function getHeight()
159 	{
160 		return (height);
161 	};
162 	
163 	this.getName=function getName()
164 	{
165 		return (shapeName);
166 	};
167 	
168 	this.getData=function getData()
169 	{
170 		return (data);
171 	};
172 	
173 	this.getColor=function getColor()
174 	{
175 		return (color);
176 	};
177 	
178 	this.getRadius=function getRadius()
179 	{
180 		return (radius);
181 	};
182 	
183 	this.getTextFont=function getTextFont()
184 	{
185 		return (textFont);
186 	};
187 	
188 	this.getTextSize=function getTextSize()
189 	{
190 		return (textSize);
191 	};
192 	
193 	this.getLineColor=function getLineColor()
194 	{
195 		return (lineColor);
196 	};
197 	
198 	this.getFillColor=function getFillColor()
199 	{
200 		return (fillColor);
201 	};
202 	
203 	this.setXPoints=function setXPoints(xPoints)
204 	{
205 		x=xPoints;
206 		canvasStyles.modifyCSSAttribute("left", x[0]+"px");
207 		canvasObject.setAttribute('style', canvasStyles.toString());
208 	};
209 	
210 	this.setYPoints=function setYPoints(yPoints)
211 	{
212 		y=yPoints;
213 		canvasStyles.modifyCSSAttribute("top", y[0]+"px");
214 		canvasObject.setAttribute('style', canvasStyles.toString());
215 	};
216 	
217 	this.setWidth=function setWidth(aWidth)
218 	{
219 		width=aWidth;
220 		
221 		if(anID != "main-canvas")
222 		{
223 			canvasObject.width=width;
224 		}
225 	};
226 	
227 	this.setHeight=function setHeight(aHeight)
228 	{
229 		height=aHeight;
230 		
231 		if(anID != "main-canvas")
232 		{
233 			canvasObject.height=height;
234 		}
235 	};
236 	
237 	this.setName=function setName(aName)
238 	{
239 		shapeName=aName;
240 	};
241 	
242 	this.setData=function setData(aData)
243 	{
244 		data=aData;
245 	};
246 	
247 	this.setFillColor=function setFillColor(aColor)
248 	{
249 		fillColor=aColor;
250 		graphicTools.setFillColor(aColor);
251 	};
252 	
253 	this.setLineColor=function setLineColor(aColor)
254 	{
255 		lineColor=aColor;
256 		graphicTools.setLineColor(aColor);
257 	};
258 	
259 	this.setDrawWidth=function setDrawWidth(aDrawWidth)
260 	{
261 		shapeLineWidth=aDrawWidth;
262 		graphicTools.setGraphicBorderWidth(shapeLineWidth);
263 	};
264 	
265 	this.setRadius=function setRadius(aRadius)
266 	{
267 		radius=aRadius;
268 	};
269 	
270 	this.setColor=function setColor(aColor)
271 	{
272 		lineColor=aColor;
273 		fillColor=aColor;
274 		
275 		graphicTools.setLineColor(lineColor);
276 		graphicTools.setFillColor(fillColor);
277 	};
278 	
279 	this.setTextFont=function setTextFont(aFont)
280 	{
281 		textFont=aFont;
282 		graphicTools.setFont(textFont);
283 	};
284 	
285 	this.setTextSize=function setTextSize(aSize)
286 	{
287 		textSize=aSize;
288 		graphicTools.setFontSize(textSize);
289 	};
290 
291 	this.modifyCanvasCSS=function modifyCanvasCSS(attrib, value)
292 	{
293 		canvasStyles.modifyCSSAttribute(attrib, value);
294 		canvasObject.setAttribute('style', canvasStyles.toString());
295 	};
296 	
297 	this.detatchCanvas=function detatchCanvas()
298 	{
299 		if(anID=="container")
300 		{
301 			document.getElementById(anID).removeChild(canvasObject);
302 		}
303 	
304 		else
305 		{
306 			document.getElementById('ctatdiv'+anID).removeChild(canvasObject);
307 		}
308 	};
309 	
310 	this.shapeMagicTrigFunctionX=function shapeMagicTrigFunctionX(pointRatio)
311 	{
312 		graphicTools.magicTrigFunctionX(pointRatio);
313 	};
314 
315 	this.shapeMagicTrigFunctionY=function shapeMagicTrigFunctionY(pointRatio)
316 	{
317 		graphicTools.magicTrigFunctionY(pointRatio);
318 	};
319 	
320 	this.drawLine=function drawLine()
321 	{
322 		graphicTools.drawLine(x[0]+shapeLineWidth, y[0]+shapeLineWidth, x[1], y[1], color, shapeLineWidth);
323 	};
324 
325 	/**
326 	 * The draw circle methods require us to begin at (initial+radius) because HTML5 assumes
327 	 * that (0, 0) is at the center of the circle, rather than at the corner.
328 	 */
329 	this.drawCircle=function drawCircle()
330 	{
331 		graphicTools.drawCircle(x[0]+radius, y[0]+radius, radius);
332 	};
333 
334 	this.drawCircleFilled=function drawCircleFilled()
335 	{
336 		graphicTools.drawCircleFilled(x[0]+radius, y[0]+radius, radius-1);
337 	};
338 
339 	this.drawRectangle=function drawRectangle(x, y, aWidth, aHeight)
340 	{
341 		graphicTools.drawRectangle(x[0]+shapeLineWidth, y[0]+shapeLineWidth, width-shapeLineWidth, height-shapeLineWidth);
342 	};
343 
344 	this.drawRectangleFilled=function drawRectangleFilled()
345 	{
346 		graphicTools.drawRectangleFilled(x[0], y[0], width, height);
347 	};
348 	
349 	this.drawRoundedRect=function drawRoundedRect ()
350 	{		
351 		graphicTools.drawRoundedRect(x[0]+shapeLineWidth, y[0]+shapeLineWidth, width-radius-shapeLineWidth, height-radius-shapeLineWidth, radius);
352 	};
353 	
354 	this.drawRoundedRectFilled=function drawRoundedRectFilled ()
355 	{
356 		graphicTools.drawRoundedRectFilled(x[0]+shapeLineWidth, y[0]+shapeLineWidth, width-radius-shapeLineWidth, height-radius-shapeLineWidth, radius);
357 	};
358 	
359 	this.drawText=function drawText ()
360 	{				
361 		graphicTools.drawText((textSize/2), (textSize/2), data);
362 	};
363 	
364 	this.drawTriangle=function drawTriangle()
365 	{
366 		graphicTools.drawTriangle(x[0]+shapeLineWidth, y[0]+shapeLineWidth, x[1], y[1], x[2], y[2]);
367 	};
368 
369 	this.drawTriangleFilled=function drawTriangleFilled()
370 	{
371 		graphicTools.drawTriangleFilled(x[0], y[0], x[1], y[1], x[2], y[2]);
372 	};
373 
374 	this.save=function save()
375 	{
376 		graphicTools.save();
377 	};
378 
379 	this.restore=function restore()
380 	{
381 		graphicTools.restore();
382 	};
383 
384 	this.clip=function clip()
385 	{
386 		graphicTools.clip();
387 	};
388 	
389 	/**
390 	 * Gives you a new canvas to draw on, but is situated in the same
391 	 * spot as the last one. Doing so effectively "erases" the painted object.
392 	 */
393 	this.erase=function erase()
394 	{
395 		pointer.detatchCanvas();
396 		x=new Array();
397 		y=new Array();
398 	
399 		canvasObject=document.createElement('canvas');
400 		canvasObject.id=shapeName;
401 		canvasObject.width=width;
402 		canvasObject.height=height;
403 		
404 		canvasStyles=new CTATCSS();
405 
406 		canvasStyles.addCSSAttribute("position", "absolute");
407 		canvasStyles.addCSSAttribute("left", canvasX+"px");
408 		canvasStyles.addCSSAttribute("top", canvasY+"px");
409 		
410 		canvasStyles.addCSSAttribute("visibility", "visible");
411 		canvasStyles.addCSSAttribute("background-color", "rgba(0, 0, 0, 0)");
412 		canvasStyles.addCSSAttribute("border-style", "none");	
413 		
414 		canvasObject.setAttribute('style', canvasStyles.toString());
415 		
416 		if(anID=="container")
417 		{
418 			document.getElementById(anID).appendChild(canvasObject);
419 		}
420 	
421 		else
422 		{
423 			document.getElementById('ctatdiv'+anID).appendChild(canvasObject);
424 		}
425 		
426 		//Add the event listeners from the last canvas
427 		for(var i=0; i < eventHandlerList.length; i++)
428 		{
429 			canvasObject.addEventListener(eventHandlerList[i][0], eventHandlerList[i][1]);
430 		}
431 	};	
432 }
