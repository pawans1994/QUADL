 21 function CTATFlashVars ()
 22 {
 23 	CTATBase.call (this, "CTATFlashVars", "__undefined__");
 24 	
 25 	var pointer=this;
 26 	var raw=null;
 27 	
 28 	/**
 29 	 * 
 30 	 * @param aData
 31 	 */
 32 	this.assignRawFlashVars=function assignRawFlashVars (aData)
 33 	{
 34 		this.raw=aData;
 35 	};
 36 	
 37 	/**
 38 	 * 
 39 	 */
 40 	this.getRawFlashVars=function getRawFlashVars ()
 41 	{
 42 		return (this.raw);
 43 	};
 44 	
 45 	/**
 46 	 * <b>[Required]</b> The local time zone name. 
 47 	 * <p>Datashop prefers the use of tz database time zones, such as one from the "TZ" column in this <a href="http://en.wikipedia.org/wiki/List_of_zoneinfo_time_zones">List of zoneinfo time zones</a>. 
 48 	 * Three-letter time zone abbreviations such as "EST" and "PST" are still valid, but are deprecated. If the field is not assigned
 49 	 * the default value of "America/New_York" will be used.</p>
 50 	 */
 51 	this.setTimeZone=function setTimeZone (zone) 
 52 	{
 53 		var getDebugger = new CTATBase("CTATFlashVars", "");
 54 
 55 		getDebugger.debug ("setTimeZone ("+zone+")");		
 56 		
 57 		var timeZone="";
 58 	
 59 		if ((zone==null) || (zone=="") || (zone==undefined)) 
 60 		{
 61 		    var tz = jstz.determine(); // Determines the time zone of the browser client
 62 		    			
 63 		    getDebugger.debug ("Assigning detected timezone: " + tz.name ());
 64 		    
 65 			timeZone = tz.name(); // Returns the name of the time zone eg "Europe/Berlin"
 66 		}
 67 		else 
 68 		{
 69 			if (zone.length > 50)
 70 				zone = zone.substr(0, 50);
 71 			
 72 			if (zone.length == 3 || zone.length == 4)
 73 			{
 74 				getDebugger.debug("3 and 4 letter time zone abbreviations are deprecated. See list of tz database zone names for better options");
 75 			}	
 76 			
 77 			timeZone = zone;
 78 		}
 79 		
 80 		if (this.raw!=null)
 81 		{
 82 			this.raw ['timezone']=timeZone;
 83 		}	
 84 	};
 85 	
 86 	/**
 87 	 * 
 88 	 * @returns
 89 	 */
 90 	this.getTimeZone=function getTimeZone() 
 91 	{
 92 		if (this.raw!=null)
 93 		{
 94 			return (this.raw ['timezone']);
 95 		}
 96 		
 97 		return "";
 98 	};
 99 }
